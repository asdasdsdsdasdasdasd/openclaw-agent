Extract odds (evaluate) — skip 即場 / 同場過關:

() => {
  const WANT = ['主客和','半場主客和','讓球','半場讓球','入球大細','半場入球大細','開出角球大細','半場開出角球大細'];
  const SKIP = ['即場','同場過關'];
  const out = {};
  // Find section headers in odds panel (#detail or 最後賠率 area)
  const root = document.querySelector('[class*="detail"]') || document.body;
  const allText = document.body.innerText;
  for (const name of WANT) {
    if (SKIP.some(s => name.includes(s))) continue;
    // Grab section: header name until next major header
    const idx = allText.indexOf(name);
    if (idx < 0) { out[name] = []; continue; }
    const slice = allText.slice(idx, idx + 2500);
    // skip if this header is 即場 variant
    if (/即場/.test(slice.slice(0, name.length + 30))) continue;
    const lines = [];
    // lines like [8.5] 1.64 or -0.5/-1 2.01 or 主隊勝 1.64
    const lineRe = /(\[[^\]]+\]|[-+]?\d+(?:\.\d+)?(?:\/[-+]?\d+(?:\.\d+)?)?)\s+(\d+\.\d+)/g;
    let m;
    while ((m = lineRe.exec(slice)) !== null) {
      lines.push({ line: m[1], odds: parseFloat(m[2]) });
    }
    // Also capture 1X2 style: 主隊勝/和/客隊勝 with odds
    if (name.includes('主客和')) {
      const hda = [...slice.matchAll(/(主隊勝|和|客隊勝)[^\d]*(\d+\.\d+)/g)].map(x => ({ selection: x[1], odds: parseFloat(x[2]) }));
      if (hda.length) lines.push(...hda.map(x => ({ line: x.selection, odds: x.odds })));
    }
    out[name] = lines;
  }
  return out;
}
