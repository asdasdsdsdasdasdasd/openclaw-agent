Step C — Scrape current page (evaluate)

() => {
  const rows = [];
  document.querySelectorAll('.fb-results-match-results .table-row').forEach(row => {
    if (row.classList.contains('table-header')) return;
    const cells = row.querySelectorAll('.table-cell');
    if (cells.length < 6) return;
    const match_id = cells[1]?.innerText?.trim();
    if (!match_id?.startsWith('FB')) return;
    rows.push({
      date: cells[0].innerText.trim(),
      match_id,
      teams: cells[3].innerText.trim(),
      ht_score: cells[4].innerText.trim(),
      ft_score: cells[5].innerText.trim()
    });
  });
  const m = document.body.innerText.match(/本頁顯示\s*(\d+)\s*-\s*(\d+)\s*場，共找到\s*(\d+)\s*場/);
  return {
    rows,
    summary: m ? m[0] : null,
    start: m ? +m[1] : null,
    end: m ? +m[2] : null,
    total: m ? +m[3] : null,
    activePage: document.querySelector('.pagination-box .active')?.innerText?.trim() || null
  };
}

Append rows to an in-memory list (dedupe by match_id).

Step D — Paginate (evaluate ONLY — NO SCROLL)

To go to page N (N = 2, 3, ...):

(n) => {
  const box = document.querySelector('.pagination-box');
  if (!box) return { ok: false, error: 'no pagination-box' };
  const target = Array.from(box.children).find(el => el.innerText.trim() === String(n));
  if (target) { target.click(); return { ok: true, action: 'clicked page ' + n }; }
  const arrow = box.querySelector('.arrow-icon-default-right');
  if (arrow && !arrow.closest('.disable')) {
    arrow.click();
    return { ok: true, action: 'clicked next arrow' };
  }
  return { ok: false, error: 'page ' + n + ' not found' };
}

Loop:
- After each page click, wait ~2s, scrape again (Step C)
- Stop when end >= total OR pagination returns ok:false
